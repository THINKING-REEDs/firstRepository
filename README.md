<template>  <div class='app'>      <input type="text" v-model="inputValue" @keyup.enter="todo">      <button @click='todo'>提交事项</button>      <hr>      <ul v-if='doingList'>          <li v-for="item in doingList" :key='item.id'>              <span>{{item.name}}</span>               <button @click="done(item.id)">完成处理</button>          </li>      </ul>      <hr>      <ul v-if='doneList'>          <li v-for="item in doneList" :key='item.id'>              <span>{{item.name}}</span>               <button @click="doAgain(item.id)">重新处理</button>          </li>      </ul>  </div></template><script>export default {  data() {    return {        inputValue: '',        list: [],    };  },  computed:{      doneList () {          return this.list.filter((item)=>{              if(item.status == 2) {                  return true              }          })      },      doingList () {          return this.list.filter((item)=>{              if(item.status == 1) {                  return true              }          })      },  },  methods:{      todo () {          this.list.push({              name: this.inputValue,              status: 1,              id: this.list.length,          })          this.inputValue = ''      },      done (id) {          this.list[id].status = 2      },      doAgain (id) {          this.list[id].status = 1      }  },}</script><style scoped></style>
